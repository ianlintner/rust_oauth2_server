26impl Default for Config {
27    fn default() -> Self {
28        Self {
29            server: ServerConfig {
30                host: std::env::var("OAUTH2_SERVER_HOST").unwrap_or_else(|_| "127.0.0.1".to_string()),
31                port: std::env::var("OAUTH2_SERVER_PORT")
32                    .ok()
33                    .and_then(|p| p.parse().ok())
34                    .unwrap_or(8080),
35            },
36            database: DatabaseConfig {
37                url: std::env::var("OAUTH2_DATABASE_URL").unwrap_or_else(|_| "sqlite:oauth2.db".to_string()),
38            },
39            jwt: JwtConfig {
40                // Use environment variable or fail-safe default for testing
41                // Production deployments MUST set OAUTH2_JWT_SECRET
42                secret: std::env::var("OAUTH2_JWT_SECRET")
43                    .unwrap_or_else(|_| {
44                        eprintln!("WARNING: OAUTH2_JWT_SECRET not set. Using insecure default for testing only!");
45                        eprintln!("NEVER use this in production! Set OAUTH2_JWT_SECRET environment variable.");
46                        "insecure-default-for-testing-only-change-in-production".to_string()
47                    }),
48            },
49        }
50    }
51}

53impl Config {
54    #[allow(dead_code)] // Planned for future environment-based configuration
55    pub fn from_env() -> Result<Self, config::ConfigError> {
56        let config = config::Config::builder()
57            .add_source(config::Environment::with_prefix("OAUTH2"))
58            .build()?;

60        config.try_deserialize()
61    }
62
63    /// Validate configuration for production use
64    pub fn validate_for_production(&self) -> Result<(), String> {
65        // Check JWT secret is not the default
66        if self.jwt.secret == "insecure-default-for-testing-only-change-in-production" {
67            return Err("OAUTH2_JWT_SECRET must be explicitly set for production. Generate a secure random string (minimum 32 characters).".to_string());
68        }
69
70        // Check JWT secret length
71        if self.jwt.secret.len() < 32 {
72            return Err(format!(
73                "OAUTH2_JWT_SECRET must be at least 32 characters long (current: {} characters)",
74                self.jwt.secret.len()
75            ));
76        }
77
78        Ok(())
79    }
