36impl SocialLoginConfig {
37    pub fn from_env() -> Self {
38        Self {
39            google: Self::provider_from_env("GOOGLE"),
40            microsoft: Self::provider_from_env("MICROSOFT"),
41            github: Self::provider_from_env("GITHUB"),
42            azure: Self::provider_from_env("AZURE"),
43            okta: Self::provider_from_env("OKTA"),
44            auth0: Self::provider_from_env("AUTH0"),
45        }
46    }
47
48    fn provider_from_env(prefix: &str) -> Option<ProviderConfig> {
49        let client_id = std::env::var(format!("OAUTH2_{}_CLIENT_ID", prefix)).ok()?;
50        let client_secret = std::env::var(format!("OAUTH2_{}_CLIENT_SECRET", prefix)).ok()?;
51        let redirect_uri =
52            std::env::var(format!("OAUTH2_{}_REDIRECT_URI", prefix)).unwrap_or_else(|_| {
53                format!(
54                    "http://localhost:8080/auth/callback/{}",
55                    prefix.to_lowercase()
56                )
57            });
58
59        Some(ProviderConfig {
60            client_id,
61            client_secret,
62            redirect_uri,
63            tenant_id: std::env::var(format!("OAUTH2_{}_TENANT_ID", prefix)).ok(),
64            domain: std::env::var(format!("OAUTH2_{}_DOMAIN", prefix)).ok(),
65        })
66    }
