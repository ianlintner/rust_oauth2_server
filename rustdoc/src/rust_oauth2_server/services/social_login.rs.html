10// Type alias for a fully configured OAuth2 client with all required endpoints set.
11// This is necessary due to oauth2 5.0's typestate pattern which tracks endpoint
12// configuration at compile time. The generic parameters represent:
13// - Error types for standard responses and revocation
14// - Token response and introspection types
15// - Endpoint states: auth URL (Set), token URL (Set), device/introspection/revocation (NotSet)
16type ConfiguredClient = oauth2::Client<
17    oauth2::StandardErrorResponse<oauth2::basic::BasicErrorResponseType>,
18    oauth2::StandardTokenResponse<oauth2::EmptyExtraTokenFields, oauth2::basic::BasicTokenType>,
19    oauth2::StandardTokenIntrospectionResponse<
20        oauth2::EmptyExtraTokenFields,
21        oauth2::basic::BasicTokenType,
22    >,
23    oauth2::StandardRevocableToken,
24    oauth2::StandardErrorResponse<oauth2::RevocationErrorResponseType>,
25    EndpointSet,
26    EndpointNotSet,
27    EndpointNotSet,
28    EndpointNotSet,
29    EndpointSet,
30>;
31

35    pub fn get_google_client(config: &ProviderConfig) -> Result<ConfiguredClient, OAuth2Error> {
36        Ok(BasicClient::new(ClientId::new(config.client_id.clone()))
37            .set_client_secret(ClientSecret::new(config.client_secret.clone()))
38            .set_auth_uri(
39                AuthUrl::new("https://accounts.google.com/o/oauth2/v2/auth".to_string())
40                    .map_err(|e| OAuth2Error::new("invalid_configuration", Some(&e.to_string())))?,
41            )
42            .set_token_uri(
43                TokenUrl::new("https://oauth2.googleapis.com/token".to_string())
44                    .map_err(|e| OAuth2Error::new("invalid_configuration", Some(&e.to_string())))?,
45            )
46            .set_redirect_uri(
47                RedirectUrl::new(config.redirect_uri.clone())
48                    .map_err(|e| OAuth2Error::new("invalid_configuration", Some(&e.to_string())))?,
49            ))
50    }

135    pub async fn fetch_google_user_info(access_token: &str) -> Result<SocialUserInfo, OAuth2Error> {
136        let client = reqwest::Client::new();
137        let response = client
138            .get("https://www.googleapis.com/oauth2/v2/userinfo")
139            .bearer_auth(access_token)
140            .send()
141            .await
142            .map_err(|e| OAuth2Error::new("provider_error", Some(&e.to_string())))?;
143
144        #[derive(Deserialize)]
145        struct GoogleUser {
146            id: String,
147            email: String,
148            name: Option<String>,
149            picture: Option<String>,
150        }
151
152        let user: GoogleUser = response
153            .json()
154            .await
155            .map_err(|e| OAuth2Error::new("provider_error", Some(&e.to_string())))?;
156
157        Ok(SocialUserInfo {
158            provider: "google".to_string(),
159            provider_user_id: user.id,
160            email: user.email,
161            name: user.name,
162            picture: user.picture,
163        })
164    }

200    pub async fn fetch_github_user_info(access_token: &str) -> Result<SocialUserInfo, OAuth2Error> {
201        let client = reqwest::Client::new();
202        let response = client
203            .get("https://api.github.com/user")
204            .bearer_auth(access_token)
205            .header("User-Agent", "rust_oauth2_server")
206            .send()
207            .await
208            .map_err(|e| OAuth2Error::new("provider_error", Some(&e.to_string())))?;
209
210        #[derive(Deserialize)]
211        struct GitHubUser {
212            id: i64,
213            email: Option<String>,
214            name: Option<String>,
215            avatar_url: Option<String>,
216        }
217
218        let user: GitHubUser = response
219            .json()
220            .await
221            .map_err(|e| OAuth2Error::new("provider_error", Some(&e.to_string())))?;
222
223        // GitHub might not provide email in the main call
224        let email = if let Some(email) = user.email {
225            email
226        } else {
227            // Fetch primary email
228            let email_response = client
229                .get("https://api.github.com/user/emails")
230                .bearer_auth(access_token)
231                .header("User-Agent", "rust_oauth2_server")
232                .send()
233                .await
234                .map_err(|e| OAuth2Error::new("provider_error", Some(&e.to_string())))?;
235
236            #[derive(Deserialize)]
237            struct GitHubEmail {
238                email: String,
239                primary: bool,
240            }
241
242            let emails: Vec<GitHubEmail> = email_response
243                .json()
244                .await
245                .map_err(|e| OAuth2Error::new("provider_error", Some(&e.to_string())))?;
246
247            emails
248                .into_iter()
249                .find(|e| e.primary)
250                .map(|e| e.email)
251                .ok_or_else(|| OAuth2Error::new("provider_error", Some("No email found")))?
252        };
253
254        Ok(SocialUserInfo {
255            provider: "github".to_string(),
256            provider_user_id: user.id.to_string(),
257            email,
258            name: user.name,
259            picture: user.avatar_url,
260        })
261    }
