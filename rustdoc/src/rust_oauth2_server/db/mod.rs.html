3use crate::models::{AuthorizationCode, Client, OAuth2Error, Token, User};
4use sqlx::{Pool, Sqlite, SqlitePool};
5
6pub struct Database {
7    pool: Pool<Sqlite>,
8}
9
10impl Database {
11    pub async fn new(database_url: &str) -> Result<Self, sqlx::Error> {
12        let pool = SqlitePool::connect(database_url).await?;
13        Ok(Self { pool })
14    }
15
16    pub async fn init(&self) -> Result<(), sqlx::Error> {
17        // With Flyway, we don't need to create tables here
18        // Just verify the connection works
19        sqlx::query("SELECT 1").execute(&self.pool).await?;
20        Ok(())
21    }

23    // Client operations
24    pub async fn save_client(&self, client: &Client) -> Result<(), OAuth2Error> {
25        sqlx::query(
26            r#"
27            INSERT INTO clients (id, client_id, client_secret, redirect_uris, grant_types, scope, name, created_at, updated_at)
28            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
29            "#,
30        )
31        .bind(&client.id)
32        .bind(&client.client_id)
33        .bind(&client.client_secret)
34        .bind(&client.redirect_uris)
35        .bind(&client.grant_types)
36        .bind(&client.scope)
37        .bind(&client.name)
38        .bind(client.created_at)
39        .bind(client.updated_at)
40        .execute(&self.pool)
41        .await?;
42        Ok(())
43    }
44
45    pub async fn get_client(&self, client_id: &str) -> Result<Option<Client>, OAuth2Error> {
46        let client = sqlx::query_as::<_, Client>("SELECT * FROM clients WHERE client_id = ?")
47            .bind(client_id)
48            .fetch_optional(&self.pool)
49            .await?;
50        Ok(client)
51    }
