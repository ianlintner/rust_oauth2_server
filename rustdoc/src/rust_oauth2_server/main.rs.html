54#[actix_web::main]
55async fn main() -> std::io::Result<()> {
56    // Initialize telemetry and tracing
57    telemetry::init_telemetry("oauth2_server").unwrap_or_else(|e| {
58        eprintln!("Failed to initialize telemetry: {}", e);
59        // Fall back to basic logging
60        env_logger::init_from_env(env_logger::Env::new().default_filter_or("info"));
61    });
62
63    tracing::info!("Starting OAuth2 Server...");
64
65    // Load configuration
66    let config = config::Config::default();
67
68    // Validate configuration for production
69    if let Err(e) = config.validate_for_production() {
70        tracing::warn!("Configuration validation warning: {}", e);
71        tracing::warn!("This configuration should only be used for testing!");
72    }
73
74    tracing::info!("Configuration loaded");
75
76    // Load social login configuration
77    let social_config = Arc::new(models::SocialLoginConfig::from_env());
78    tracing::info!("Social login configuration loaded");
79
80    // Initialize metrics
81    let metrics = metrics::Metrics::new().expect("Failed to initialize metrics");
82    tracing::info!("Metrics initialized");
83
84    // Initialize database
85    let db = db::Database::new(&config.database.url)
86        .await
87        .expect("Failed to connect to database");
88
89    db.init().await.expect("Failed to initialize database");
90    tracing::info!("Database initialized");
91
92    let db = Arc::new(db);
93    let jwt_secret = config.jwt.secret.clone();
94
95    // Load session key from environment or generate a new one
96    // In production, OAUTH2_SESSION_KEY should be set to a persistent value
97    let session_key = if let Ok(key_str) = std::env::var("OAUTH2_SESSION_KEY") {
98        if key_str.len() < 64 {
99            panic!("OAUTH2_SESSION_KEY must be at least 64 characters (128 hex digits)");
100        }
101        let key_bytes =
102            hex::decode(&key_str).expect("OAUTH2_SESSION_KEY must be valid hexadecimal");
103        Key::try_from(&key_bytes[..]).expect("OAUTH2_SESSION_KEY must be exactly 64 bytes")
104    } else {
105        tracing::warn!("OAUTH2_SESSION_KEY not set. Generating random key. Sessions will not persist across restarts!");
106        Key::generate()
107    };
108
109    // Start actors
110    let token_actor = actors::TokenActor::new(db.clone(), jwt_secret.clone()).start();
111    let client_actor = actors::ClientActor::new(db.clone()).start();
112    let auth_actor = actors::AuthActor::new(db.clone()).start();
113
114    tracing::info!("Actors started");
115
116    // OpenAPI documentation
117    let openapi = ApiDoc::openapi();
118
119    let bind_addr = format!("{}:{}", config.server.host, config.server.port);
120    tracing::info!("Starting server at http://{}", bind_addr);
121    tracing::info!("Login page available at http://{}/auth/login", bind_addr);
122    tracing::info!("Swagger UI available at http://{}/swagger-ui", bind_addr);
123    tracing::info!("Admin dashboard at http://{}/admin", bind_addr);
124    tracing::info!("Metrics endpoint at http://{}/metrics", bind_addr);
125
126    // Start HTTP server
127    let server = HttpServer::new(move || {
128        let cors = Cors::default()
129            .allow_any_origin()
130            .allow_any_method()
131            .allow_any_header()
132            .max_age(3600);
133
134        App::new()
135            // Middleware
136            .wrap(SessionMiddleware::new(
137                CookieSessionStore::default(),
138                session_key.clone(),
139            ))
140            .wrap(TracingLogger::default())
141            .wrap(actix_middleware::Logger::default())
142            .wrap(actix_middleware::Compress::default())
143            .wrap(middleware::MetricsMiddleware::new(metrics.clone()))
144            .wrap(cors)
145            // Shared state
146            .app_data(web::Data::new(token_actor.clone()))
147            .app_data(web::Data::new(client_actor.clone()))
148            .app_data(web::Data::new(auth_actor.clone()))
149            .app_data(web::Data::new(jwt_secret.clone()))
150            .app_data(web::Data::new(db.clone()))
151            .app_data(web::Data::new(metrics.clone()))
152            .app_data(web::Data::new(social_config.clone()))
153            // Root route
154            .route(
155                "/",
156                web::get().to(|| async {
157                    HttpResponse::Found()
158                        .append_header(("Location", "/auth/login"))
159                        .finish()
160                }),
161            )
162            // Authentication routes
163            .service(
164                web::scope("/auth")
165                    .route("/login", web::get().to(handlers::auth::login_page))
166                    .route("/logout", web::post().to(handlers::auth::logout))
167                    .route("/success", web::get().to(handlers::auth::auth_success))
168                    .service(
169                        web::scope("/login")
170                            .route("/google", web::get().to(handlers::auth::google_login))
171                            .route("/microsoft", web::get().to(handlers::auth::microsoft_login))
172                            .route("/github", web::get().to(handlers::auth::github_login))
173                            .route("/azure", web::get().to(handlers::auth::microsoft_login)) // Azure uses Microsoft endpoint
174                            // NOTE: Okta and Auth0 handlers not yet implemented - buttons should be hidden in UI
175                            // or implement proper handlers in handlers::auth module
176                            .route(
177                                "/okta",
178                                web::get().to(|| async {
179                                    actix_web::HttpResponse::ServiceUnavailable()
180                                        .body("Okta login not yet implemented")
181                                }),
182                            )
183                            .route(
184                                "/auth0",
185                                web::get().to(|| async {
186                                    actix_web::HttpResponse::ServiceUnavailable()
187                                        .body("Auth0 login not yet implemented")
188                                }),
189                            ),
190                    )
191                    .route(
192                        "/callback/{provider}",
193                        web::get().to(handlers::auth::auth_callback),
194                    ),
195            )
196            // OAuth2 endpoints
197            .service(
198                web::scope("/oauth")
199                    .route("/authorize", web::get().to(handlers::oauth::authorize))
200                    .route("/token", web::post().to(handlers::oauth::token))
201                    .route("/introspect", web::post().to(handlers::token::introspect))
202                    .route("/revoke", web::post().to(handlers::token::revoke)),
203            )
204            // Client management endpoints
205            .service(web::scope("/clients").route(
206                "/register",
207                web::post().to(handlers::client::register_client),
208            ))
209            // Well-known endpoints
210            .service(web::scope("/.well-known").route(
211                "/openid-configuration",
212                web::get().to(handlers::wellknown::openid_configuration),
213            ))
214            // Admin endpoints
215            .service(
216                web::scope("/admin")
217                    .route("", web::get().to(admin_dashboard))
218                    .service(
219                        web::scope("/api")
220                            .route("/dashboard", web::get().to(handlers::admin::dashboard))
221                            .route("/clients", web::get().to(handlers::admin::list_clients))
222                            .route("/tokens", web::get().to(handlers::admin::list_tokens))
223                            .route(
224                                "/tokens/{id}/revoke",
225                                web::post().to(handlers::admin::admin_revoke_token),
226                            )
227                            .route(
228                                "/clients/{id}",
229                                web::delete().to(handlers::admin::delete_client),
230                            ),
231                    ),
232            )
233            // Error page
234            .route("/error", web::get().to(error_page))
235            // Observability endpoints
236            .route("/health", web::get().to(handlers::admin::health))
237            .route("/ready", web::get().to(handlers::admin::readiness))
238            .route("/metrics", web::get().to(handlers::admin::system_metrics))
239            // Swagger UI
240            .service(
241                SwaggerUi::new("/swagger-ui/{_:.*}").url("/api-docs/openapi.json", openapi.clone()),
242            )
243            // Static files
244            .service(Files::new("/static", "./static"))
245    })
246    .bind(&bind_addr)?
247    .run();
248
249    server.await?;
250
251    // Shutdown telemetry
252    telemetry::shutdown_telemetry();
253
254    Ok(())
255}
