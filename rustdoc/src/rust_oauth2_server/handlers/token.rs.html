7#[derive(Debug, Deserialize)]
8pub struct IntrospectRequest {
9    token: String,
10    #[allow(dead_code)] // OAuth2 spec field, can be used for optimization
11    token_type_hint: Option<String>,
12}
13
14/// Token introspection endpoint
15/// Returns information about a token
16pub async fn introspect(
17    form: web::Form<IntrospectRequest>,
18    token_actor: web::Data<Addr<TokenActor>>,
19    jwt_secret: web::Data<String>,
20) -> Result<HttpResponse, OAuth2Error> {
21    // Try to validate the token
22    let token_result = token_actor
23        .send(ValidateToken {
24            token: form.token.clone(),
25        })
26        .await
27        .map_err(|e| OAuth2Error::new("server_error", Some(&e.to_string())))?;
28
29    match token_result {
30        Ok(token) => {
31            // Decode JWT to get claims
32            let claims = Claims::decode(&token.access_token, &jwt_secret).ok();
33
34            let response = IntrospectionResponse {
35                active: token.is_valid(),
36                scope: Some(token.scope),
37                client_id: Some(token.client_id),
38                username: Some(token.user_id.clone()),
39                token_type: Some(token.token_type),
40                exp: claims.as_ref().map(|c| c.exp),
41                iat: claims.as_ref().map(|c| c.iat),
42                sub: Some(token.user_id),
43            };
44
45            Ok(HttpResponse::Ok().json(response))
46        }
47        Err(_) => {
48            // Token is invalid
49            let response = IntrospectionResponse {
50                active: false,
51                scope: None,
52                client_id: None,
53                username: None,
54                token_type: None,
55                exp: None,
56                iat: None,
57                sub: None,
58            };
59            Ok(HttpResponse::Ok().json(response))
60        }
61    }
62}
